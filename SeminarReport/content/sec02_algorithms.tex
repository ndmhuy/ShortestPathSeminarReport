\section{Main Data Structures \& Algorithms}
In a weighted graph, the objective of a shortest path algorithm is to identify a sequence of edges connecting vertices such that the sum of the edge weights is minimized. The problem is generally bifurcated into two primary categories:
\begin{itemize}
    \item \textbf{Single Source Shortest Path (SSSP):} Determines the minimum cost to travel from a single starting node (the source) to all other reachable nodes in the graph.
    \item \textbf{All Pairs Shortest Path (APSP):} Computes the shortest paths between every possible pair of vertices simultaneously.
\end{itemize}

\section{Single Source Shortest Path (SSSP)}
The Single Source Shortest Path (SSSP) problem asks a fundamental question: \textit{What is the shortest path from one starting node to all other nodes?}

\subsection{Shortest Path in DAGs}
\textbf{Historical Context:} While shortest path traversal on Directed Acyclic Graphs represents a fundamental application of graph theory rather than a single named algorithm, its core mechanism—Topological Sorting—was mathematically formalized by Arthur B. Kahn in 1962. The sequential evaluation method used to find shortest paths has direct roots in the PERT (Program Evaluation and Review Technique) charts developed by the United States Navy in the 1950s for project scheduling.

\vspace{0.2cm}
A \textbf{Directed Acyclic Graph (DAG)} is a directed graph which forms no cycles or loops. The SSSP problem can be solved efficiently on a DAG due to the fact that the nodes can be ordered in a topological ordering via topsort and processed sequentially.

\subsubsection{Topological Order}
A \textit{linear ordering of its vertices such that for every directed edge from vertex $u$ to vertex $v$ ($u \to v$), vertex $u$ comes before vertex $v$ in the ordering}.

Topological sorting can be implemented in two standard ways: a DFS-based approach or a BFS-based approach (Kahn's algorithm). Kahn's algorithm is especially useful when we want built-in DAG validation, because it can detect non-DAG inputs when not all vertices can be processed. In this implementation, we use the DFS-based method and assume the input graph is already a DAG.

\textbf{Topological Sort Pseudocode:}
\begin{algorithm}[H]
\caption{Topological Sort}
\begin{algorithmic}[1]
\Function{topologicalSort}{$graph$}
    \State $visited \gets [\text{false}, \text{false}, \dots , \text{false}]$ \Comment{size $V$}
    \State $order \gets \text{empty stack}$
    \For{$i \gets 0$ \textbf{to} $V - 1$}
        \If{\textbf{not} $visited[i]$}
            \State \Call{dfs}{$graph, i, visited, order$}
        \EndIf
    \EndFor
    \State \Return $order$
\EndFunction

\Function{dfs}{$graph, index, visited, order$}
    \State $visited[index] \gets \text{true}$
    \For{$edge$ \textbf{in} $graph[index]$}
        \If{\textbf{not} $visited[edge.to]$}
            \State \Call{dfs}{$graph, edge.to, visited, order$}
        \EndIf
    \EndFor
    \State $order.\text{push}(index)$
\EndFunction
\end{algorithmic}
\end{algorithm}

\subsubsection{DAG Shortest Path Processing}
\begin{enumerate}
    \item Select the first coming vertex from the topological ordering.
    \item Explore all its neighbours.
    \item If finding a shorter path, relax the edge.
    \item Reprocess until finishing.
\end{enumerate}

\textbf{DAG Shortest Path Pseudocode:}
\begin{algorithm}[H]
\caption{DAG Shortest Path}
\begin{algorithmic}[1]
\Function{dagShortestPath}{$graph, start$}
    \State $distances \gets [\infty, \infty, \dots , \infty]$ \Comment{size $V$}
    \State $distances[start] \gets 0$
    \State $topoOrder \gets \Call{topologicalSort}{graph}$
    \While{$topoOrder$ \textbf{is not} empty}
        \State $index \gets topoOrder.\text{pop}()$
        \If{$distances[index] == \infty$}
            \State \textbf{continue}
        \EndIf
        \For{$edge$ \textbf{in} $graph[index]$}
            \State $newDist \gets distances[index] + edge.cost$
            \If{$newDist < distances[edge.to]$}
                \State $distances[edge.to] \gets newDist$
            \EndIf
        \EndFor
    \EndWhile
    \State \Return $distances$
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O(V + E)}$}. We are traversing every vertex and edge.
    \item \textbf{Space Complexity: $\mathbf{O(V)}$}. We are storing the topological ordering and the distances array.
\end{itemize}

\subsection{Dijkstra's Algorithm}
\textbf{Historical Context:} Conceived by the Dutch computer scientist Edsger W. Dijkstra in 1956 (and formally published in 1959 in \textit{A note on two problems in connexion with graphs}), this algorithm was famously designed in just twenty minutes while Dijkstra was drinking coffee at a café in Amsterdam. He initially created it to find the shortest path between the cities of Rotterdam and Groningen to demonstrate the capabilities of the new ARMAC computer.

\vspace{0.2cm}
Given a graph of non-negative weight edges. We can greedily assume that adding edges can only increase the total path cost. 

Therefore, we come up with the \textbf{Greedy strategy}: continually select the unvisited node with the smallest known distance using a min-heap assuming it cannot be improved anymore, and then relaxes its outgoing edges.

\textbf{Shortest Path Pseudocode:}
\begin{algorithm}[H]
\caption{Dijkstra's Algorithm}
\begin{algorithmic}[1]
\Function{dijkstra}{$graph, start$}
    \State $visited \gets [\text{false}, \text{false}, \dots , \text{false}]$ \Comment{size $V$}
    \State $distances \gets [\infty, \infty, \dots , \infty]$ \Comment{size $V$}
    \State $distances[start] \gets 0$
    \State $pq \gets \text{empty priority queue (min-heap)}$
    \State $pq.\text{insert}(\{0, start\})$
    
    \While{$pq$ \textbf{is not} empty}
        \State $index, minValue \gets pq.\text{poll}()$
        \State $visited[index] \gets \text{true}$
        \If{$distances[index] < minValue$}
            \State \textbf{continue}
        \EndIf
        \For{$edge$ \textbf{in} $graph[index]$}
            \If{$visited[edge.to]$}
                \State \textbf{continue}
            \EndIf
            \State $newDist \gets distances[index] + edge.cost$
            \If{$newDist < distances[edge.to]$}
                \State $distances[edge.to] \gets newDist$
                \State $pq.\text{insert}(\{edge.to, newDist\})$
            \EndIf
        \EndFor
    \EndWhile
    \State \Return $distances$
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O((V+E)\log V)}$}. We extract exactly $V$ vertices. Each extraction from a heap takes $O(\log V)$. We evaluate at most $E$ edges. If we relax the edge, we push new value into the heap takes $O(\log V)$.
    \item \textbf{Space Complexity: $\mathbf{O(V + E)}$}. We are storing the distances array for every vertex and the priority queue for each relaxed edge.
\end{itemize}

\subsubsection{Dijkstra Optimizing}
\begin{itemize}
    \item \textbf{Lazy Dijkstra:} uses a standard Binary Heap. Time complexity: $\mathbf{O((V + E)\log V)}$.
    \item \textbf{Eager Dijkstra:} uses an indexed priority queue to instantly locate the vertex and update the value dynamically. Time complexity: $\mathbf{O((V + E)\log V)}$.
    \item \textbf{D-ary Heap Dijkstra (dense graph optimizer):} uses a Heap giving each parent $D$ children, cheaper when updating but more expensive when polling. \\ Time complexity: $\mathbf{O(V \log_D V + E \log_D V)}$.
    \item \textbf{Fibonacci Heap Dijkstra:} Replaces the standard tree-based heap with a highly complex collection of trees called a Fibonacci Heap which achieved updating time an amortized $O(1)$. Time Complexity: $\mathbf{O(V \log V + E)}$.
\end{itemize}

\textbf{Phase 3: Finalizing Node and Selecting the Next Best Path}
\begin{enumerate}
    \setcounter{enumi}{6}
    \item \textbf{Closing the Node:} Once all neighbors of A are checked, Node A is completed. It is removed from the \texttt{Open Set} and moved to the \texttt{Closed Set}. We are now guaranteed that the shortest path to A has been found.
    \item \textbf{Next Iteration:} The algorithm returns to the \texttt{Open Set} to find the node with the current minimum distance. Comparing B (Dist: 5) and C (Dist: 2), Node C is clearly the smallest.
    \item \textbf{Expansion:} Node C will be selected as the next "Current Node" in the following step. The process will repeat: checking C's neighbor (Node E), calculating cumulative distances, and updating the tables until the target is reached or all nodes are closed.
\end{enumerate}
\subsection{Shortest Path in Bellman-Ford}
\textbf{Historical Context:} This algorithm primarily bears the names of Richard Bellman (who published it in 1958 in \textit{On a routing problem}) and Lester R. Ford Jr. (who developed the foundational flow formulations in 1956). However, the algorithm was independently published by Edward F. Moore in 1959 and Alfonso Shimbel in 1955. As a pioneering application of Bellman's principle of optimality, it introduced dynamic programming to the realm of graph routing.

\vspace{0.2cm}
When a graph contains negative weight edges, Dijkstra's greedy logic break down. What should we do?

Bellman-Ford is a Single Source Shortest Path (SSSP) algorithm designed specifically to handle graphs with negative weights. Unlike Dijkstra (Greedy), Bellman-Ford uses \textbf{Dynamic Programming}. It calculates shortest paths in a bottom-up manner by gradually allowing paths to use more and more edges.

\subsubsection{How Bellman-Ford works: Global Relaxation}
\begin{itemize}
    \item \textbf{Step 1: Initialization.} Set the distance to the source node to $0$ and all other nodes to $\infty$.
    \item \textbf{Step 2: Edge Relaxation.} Iterate through every single edge $(u, v)$ with weight $w$ in the graph. If a new shorter path is found, then relax that edge.
    \item \textbf{Step 3: Repeat.} Perform this full-graph relaxation exactly $\mathbf{V - 1}$ \textbf{times}.
\end{itemize}

\textit{Why V - 1 times?} We expect after one loop the path is extended by at least one edge. The shortest path in a graph with $V$ vertices without cycles, contains at least $V - 1$ edges.

\subsubsection{Bellman-Ford Superpower: Negative Cycle Detection}
\textbf{Negative cycle:} \textit{A loop in the graph where the sum of the edge weights is less than zero. If one exists, you can loop infinitely to achieve a distance of $-\infty$. No shortest path can be defined.}

After running $V - 1$ iterations, run the relaxation one more time. If any edge still can be relaxed, that edge is in a negative cycle. All vertices can be routed from the negative vertices have $-\infty$ distance.

\textbf{Bellman-Ford Pseudocode:}
\begin{algorithm}[H]
\caption{Bellman-Ford Algorithm}
\begin{algorithmic}[1]
\Function{bellmanford}{$graph, start$}
    \State $dist \gets [\infty, \infty, \dots , \infty]$ \Comment{size $V$}
    \State $dist[start] \gets 0$
    \For{$i \gets 1$ \textbf{to} $V - 1$}
        \For{$edge$ \textbf{in} $graph.edges$}
            \If{$dist[edge.from] + edge.cost < dist[edge.to]$}
                \State $dist[edge.to] \gets dist[edge.from] + edge.cost$
            \EndIf
        \EndFor
    \EndFor
    
    \State $negVertices \gets \text{empty queue}$
    \For{$edge$ \textbf{in} $graph.edges$}
        \If{$dist[edge.from] + edge.cost < dist[edge.to]$}
            \State $dist[edge.to] \gets -\infty$
            \State $negVertices.\text{enqueue}(edge.to)$
        \EndIf
    \EndFor
    
    \State \Call{processNegativeCycle}{$graph, negVertices, dist$}
    \State \Return $dist$
\EndFunction

\Function{processNegativeCycle}{$graph, negVertices, dist$}
    \State $visited \gets [\text{false}, \text{false}, \dots , \text{false}]$ \Comment{size $V$}
    \While{$negVertices$ \textbf{is not} empty}
        \State $vertex \gets negVertices.\text{dequeue}()$
        \If{$visited[vertex]$}
            \State \textbf{continue}
        \EndIf
        \State \Call{dfs}{$graph, vertex, negVertices, dist, visited$}
    \EndWhile
\EndFunction

\Function{dfs}{$graph, vertex, negVertices, dist, visited$}
    \State $visited[vertex] \gets \text{true}$
    \For{$edge$ \textbf{in} $graph[vertex]$}
        \If{$visited[edge.to]$}
            \State \textbf{continue}
        \EndIf
        \State $dist[edge.to] \gets -\infty$
        \State $negVertices.\text{enqueue}(edge.to)$
        \State \Call{dfs}{$graph, edge.to, negVertices, dist, visited$}
    \EndFor
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O(E \cdot V)}$}. We iterate every edges $V - 1$ times and one more loop for negative cycle detection.
    \item \textbf{Space Complexity: $\mathbf{O(V)}$}. We are storing the distances array for every vertex and the priority queue for each relaxed edge.
\end{itemize}

Based on the specific edge list traversal order, the first iteration proceeds as follows:
\begin{itemize}

    \item \textbf{Domino Effect (Value Propagation):}
    \begin{itemize}
        \item \texttt{A $\to$ C} (weight 2): Using A's new value ($10$), $10 + 2 = 12$. Distance to C becomes $12$.
        \item \texttt{C $\to$ B} (weight -2): Using C's new value ($12$), $12 - 2 = 10$. Distance to B becomes $10$.
        \item \texttt{E $\to$ D} (weight 1): Using E's value ($8$), $8 + 1 = 9$. Distance to D becomes $9$.
    \end{itemize}
\end{itemize}

\subsection{SSSP Algorithm: How to track the path?}
By introducing a \texttt{prev} array, where \texttt{prev[i]} denotes its previous vertex. We simply update it whenever relaxing the edges:

\begin{algorithm}[H]
\begin{algorithmic}[1]
\If{$dist[edge.from] + edge.cost < dist[edge.to]$}
    \State $dist[edge.to] \gets dist[edge.from] + edge.cost$
    \State $prev[edge.to] \gets edge.from$
\EndIf
\end{algorithmic}
\end{algorithm}

Then when we need to track the path, we can route from the end to start through the \texttt{prev} array then reversing that result.

\section{All Pairs Shortest Path (APSP)}
\textit{What is the shortest path between all pairs of nodes?}

\subsection{Floyd-Warshall Algorithm}
\textbf{Historical Context:} Published independently by Robert W. Floyd (\textit{Algorithm 97: Shortest Path}, 1962) and Stephen Warshall (\textit{A theorem on Boolean matrices}, 1962), this algorithm is an elegant realization of dynamic programming. It fundamentally relies on a theorem Warshall initially developed for computing the transitive closure of a graph, which Floyd ingeniously adapted to compute minimal path weights.

\vspace{0.2cm}
We are no longer looking for paths from a single source. We want to find the shortest path between every pair of vertices $(u, v)$ simultaneously.

Floyd-Warshall is an APSP dynamic programming algorithm which relies on the idea of comparing between direct connection and routing through an intermediate vertex.

\subsubsection{The Intermediate Node}
\textit{Can we find a shorter path from $i$ to $j$ by routing it through an intermediate node $k$?}

\begin{itemize}
    \item \textbf{Initialization:} Create a 2D matrix where \texttt{dist[i][j]} is the weight of the direct edge from $i$ to $j$ (or $\infty$ if no direct edge exists).
    \item We iterate $V$ loops with index of $k$, that $\texttt{dist}[i][k] + \texttt{dist}[k][j]$ is the distance when routing through vertex $k$. If it is better than the current connection, we update $\texttt{dist}[i][j]$. We are processing for every pair $(i, j)$.
\end{itemize}

\textbf{Floyd-Warshall Pseudocode:}
\begin{algorithm}[H]
\caption{Floyd-Warshall Algorithm}
\begin{algorithmic}[1]
\Function{setup}{$graph, dist$}
    \For{$i \gets 0$ \textbf{to} $V - 1$}
        \State $dist[i][i] \gets 0$
    \EndFor
    \For{$edge$ \textbf{in} $graph.edges$}
        \State $dist[edge.from][edge.to] \gets edge.cost$
    \EndFor
\EndFunction

\Function{floydwarshall}{$graph$}
    \State $dist \gets \text{2D array of size } V \times V \text{ initialized to } \infty$
    \State \Call{setup}{$graph, dist$}
    \For{$k \gets 0$ \textbf{to} $V - 1$}
        \For{$i \gets 0$ \textbf{to} $V - 1$}
            \For{$j \gets 0$ \textbf{to} $V - 1$}
                \If{$dist[i][k] + dist[k][j] < dist[i][j]$}
                    \State $dist[i][j] \gets dist[i][k] + dist[k][j]$
                \EndIf
            \EndFor
        \EndFor
    \EndFor
    \State \Return $dist$
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}

\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O(V^3)}$}. We iterate $V$ loops to update for every pair of $(i, j)$.
    \item \textbf{Space Complexity: $\mathbf{O(V^2)}$}. We only use a 2D matrix to store the distance between every pair of vertices.
\end{itemize}

\subsubsection{How to track the path?}
By introducing a \texttt{next} 2D matrix, where \texttt{next[i][j]} denotes the next node from $i$ when routing to $j$. We initialize \texttt{next[i][j] = j} if there is a direct connection or \texttt{next[i][j] = -1}. After initializing, in the main loop, we update the \texttt{next} matrix as well as the distance:

\begin{algorithm}[H]
\begin{algorithmic}[1]
\If{$dist[i][k] + dist[k][j] < dist[i][j]$}
    \State $dist[i][j] \gets dist[i][k] + dist[k][j]$
    \State $next[i][j] \gets k$
\EndIf
\end{algorithmic}
\end{algorithm}
Then when we need to track the path, we can route from the start to end through the \texttt{next} matrix.
\subsubsection{Step-by-Step Demonstration: Floyd-Warshall Algorithm}
\textit{This section describes the step-by-step execution of the Floyd-Warshall algorithm as captured in our visualizer, demonstrating the dynamic programming matrix updates.}

\vspace{0.2cm}
\textbf{Phase 1: Initialization}
\begin{enumerate}
    \item \textbf{Matrix Setup:} The system initializes a 2D distance matrix \texttt{dist} of size $V \times V$ (for this 4-node graph, a $4 \times 4$ matrix).
    \item \textbf{Self-Distances:} The main diagonal is set to 0 (\texttt{dist[i][i] = 0}).
    \item \textbf{Direct Edges:} For every direct edge from $i$ to $j$ with weight $w$, the matrix is updated to \texttt{dist[i][j] = w}. For example, \texttt{dist[0][1] = 2} and \texttt{dist[3][0] = -2}.
    \item \textbf{No Direct Path:} Pairs of nodes without a direct connecting edge are initialized to infinity ($\infty$). For example, \texttt{dist[0][2] = $\infty$}.
\end{enumerate}

\textbf{Phase 2: Relaxation via Intermediate Node ($k = 0$)}
\begin{enumerate}
    \setcounter{enumi}{4}
    \item \textbf{The Outer Loop:} The algorithm selects each node $k$ (from $0$ to $V-1$) to act as an intermediate routing node. At each step $k$, it checks if routing from $i$ to $j$ through $k$ is shorter using the formula:
    \[ dist[i][j] = \min(dist[i][j], dist[i][k] + dist[k][j]) \]
    \item \textbf{Processing Node 0:} We analyze the first iteration where the intermediate node is \textbf{Node 0} ($k=0$). \textit{(Note: While the UI title says 'Vertex 1' due to 1-based numbering conventions, the mathematical logic is processing index 0).} Row 0 and Column 0 act as fixed references for this loop.
    \item \textbf{Case 1 - No Update:} The algorithm evaluates the cell \texttt{dist[1][1]}. The path routing through Node 0 is calculated as \texttt{dist[1][0] + dist[0][1] = 3 + 2 = 5}. Since $5 > 0$ (the current recorded distance), the value is discarded and not updated.
\end{enumerate}

\begin{enumerate}
    \setcounter{enumi}{7}
    \item \textbf{Case 2 - Successful Relaxation:} The algorithm evaluates \texttt{dist[3][1]} (path from 3 to 1).
    \begin{itemize}
        \item Current distance: \texttt{dist[3][1] = 6}
        \item New distance (via Node 0): \texttt{dist[3][0] + dist[0][1] = -2 + 2 = 0}
    \end{itemize}
    Since $0 < 6$, the path through Node 0 is strictly shorter. The system immediately updates \texttt{dist[3][1] = 0} (highlighted in blue on the UI).
    \item \textbf{Further Updates:} Similarly, the cell \texttt{dist[1][3]} is updated from $\infty$ to $6$ because \texttt{dist[1][0] + dist[0][3] = 3 + 3 = 6}. This concludes the $k=0$ loop, yielding an optimized matrix.
\end{enumerate}

\subsection{A* Search Algorithm}
\textbf{Historical Context:} Developed in 1968 by Peter Hart, Nils Nilsson, and Bertram Raphael at the Stanford Research Institute, A* was originally designed to help the Shakey robot navigate through rooms containing obstacles.

\vspace{0.2cm}
While Dijkstra's algorithm searches equally in all directions, A* uses a \textbf{heuristic} to guide the search direction towards the target, drastically reducing the number of explored nodes. It evaluates nodes by combining two metrics:
\[ f(n) = g(n) + h(n) \]
\begin{itemize}
    \item $g(n)$: The exact cost of the path from the starting node to node $n$ (identical to Dijkstra).
    \item $h(n)$: The heuristic estimated cost from node $n$ to the goal (e.g., Euclidean or Manhattan distance).
    \item $f(n)$: The total estimated cost of path through node $n$.
\end{itemize}

\textbf{A* Search Pseudocode:}
\begin{algorithm}[H]
\caption{A* Search Algorithm}
\begin{algorithmic}[1]
\Function{AStar}{$graph, start, goal$}
    \State $gScore \gets [\infty, \dots, \infty]$
    \State $gScore[start] \gets 0$
    \State $pq \gets \text{empty priority queue}$
    \State $pq.\text{insert}(\{0 + \Call{h}{start, goal}, start\})$ \Comment{\{fScore, node\}}
    
    \While{$pq$ \textbf{is not} empty}
        \State $current \gets pq.\text{poll}()$
        \If{$current == goal$}
            \State \Return $gScore[current]$
        \EndIf
        \For{$edge$ \textbf{in} $graph[current]$}
            \State $tentative\_gScore \gets gScore[current] + edge.cost$
            \If{$tentative\_gScore < gScore[edge.to]$}
                \State $gScore[edge.to] \gets tentative\_gScore$
                \State $fScore \gets tentative\_gScore + \Call{h}{edge.to, goal}$
                \State $pq.\text{insert}(\{fScore, edge.to\})$
            \EndIf
        \EndFor
    \EndWhile
\EndFunction
\end{algorithmic}
\end{algorithm}

\subsubsection{Step-by-Step Demonstration: A*}
\begin{enumerate}
    \item \textbf{Step 1 (Heuristic Initialization):} The visualizer displays the target node. Each node calculates and shows its $h(n)$ value (straight-line distance to the target).
    \item \textbf{Step 2 (Guided Search):} Unlike Dijkstra's circular expansion, A* extracts nodes based on the lowest $f(n) = g(n) + h(n)$. The visualizer shows the search "pulling" preferentially towards the goal node.
    \item \textbf{Step 3 (Path Discovery):} The algorithm terminates the moment the goal node is extracted from the priority queue, avoiding unnecessary exploration of the opposite side of the graph.
\end{enumerate}
\subsection{Johnson's Algorithm}
\textbf{Historical Context:} Published by Donald B. Johnson in 1977, this algorithm was developed to efficiently solve the All-Pairs Shortest Path (APSP) problem on sparse graphs. While Floyd-Warshall is simple, its $O(V^3)$ complexity makes it prohibitively slow for graphs with few edges. Johnson ingeniously combined Bellman-Ford and Dijkstra's algorithms, utilizing a mathematical technique to eliminate negative weights without altering the shortest paths.

\vspace{0.2cm}
For graphs with negative edges, we cannot simply use Dijkstra. We also cannot arbitrarily add a constant to all edges, as that disproportionately penalizes paths with more edges. Johnson's Algorithm solves this via a technique called \textbf{Edge Reweighting}.

\subsubsection{How Johnson's Algorithm works}
\begin{enumerate}
    \item \textbf{Add a Dummy Node:} Insert a new virtual node $S$ into the graph with $0$-weight directed edges connecting to all existing nodes.
    \item \textbf{Calculate Potentials:} Run Bellman-Ford starting from $S$ to find the minimum distance $h(v)$ to each node. This step also detects if the graph has a negative cycle.
    \item \textbf{Reweight Edges:} Adjust the weight of every original edge $(u, v)$ using the calculated potentials: 
    \[ w'(u, v) = w(u, v) + h(u) - h(v) \]
    This guarantees that all $w'(u, v) \ge 0$.
    \item \textbf{Run Dijkstra:} With all non-negative edges, execute Dijkstra's algorithm from every vertex to find the shortest paths, then reverse the reweighting formula to get the true distances.
\end{enumerate}

\textbf{Johnson's Pseudocode:}
\begin{algorithm}[H]
\caption{Johnson's Algorithm}
\begin{algorithmic}[1]
\Function{johnson}{$graph$}
    \State $G' \gets graph \text{ with a new dummy vertex } S$
    \For{$v \gets 0$ \textbf{to} $V - 1$}
        \State $G'.\text{addEdge}(S, v, 0)$
    \EndFor
    
    \State $h \gets \Call{bellmanford}{G', S}$ \Comment{Calculates potentials $h(v)$}
    \If{$h$ indicates a negative cycle}
        \State \Return "Negative cycle detected"
    \EndIf
    
    \State $rew\_graph \gets \text{empty graph}$
    \For{$edge$ \textbf{in} $graph.edges$}
        \State $w' \gets edge.cost + h[edge.from] - h[edge.to]$
        \State $rew\_graph.\text{addEdge}(edge.from, edge.to, w')$
    \EndFor
    
    \State $dist \gets \text{2D array of size } V \times V$
    \For{$u \gets 0$ \textbf{to} $V - 1$}
        \State $dijkstra\_dist \gets \Call{dijkstra}{rew\_graph, u}$
        \For{$v \gets 0$ \textbf{to} $V - 1$}
            \State $dist[u][v] \gets dijkstra\_dist[v] - h[u] + h[v]$ \Comment{Restore true distances}
        \EndFor
    \EndFor
    
    \State \Return $dist$
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O(V \cdot E \log V)}$}. We run Bellman-Ford once $O(V \cdot E)$ and Dijkstra $V$ times. Using a binary heap, $V \times O((V+E) \log V) = O(VE \log V)$ for sparse graphs.
    \item \textbf{Space Complexity: $\mathbf{O(V^2 + E)}$}. We need $O(V+E)$ for the modified graph and $O(V^2)$ to store the final all-pairs distance matrix.
\end{itemize}

\subsubsection{Step-by-Step Demonstration: Johnson's Algorithm}
\begin{enumerate}
    \item \textbf{Step 1 (Dummy Node Setup):} A new node $S$ appears on the screen, instantly projecting 0-weight edges to every other node in the graph.
    \item \textbf{Step 2 (Potential Calculation):} Bellman-Ford runs in the background, updating each node's $h(v)$ value.
    \item \textbf{Step 3 (Edge Reweighting):} The visualizer iterates through all edges. The edge values visually shift from negative/mixed values to strictly non-negative values using the $w'$ formula.
    \item \textbf{Step 4 (Dijkstra Sweep):} Dijkstra's algorithm rapidly executes sequentially from node 1, then node 2, etc., populating the final APSP matrix efficiently.
\end{enumerate}

\section{Multi-Source Shortest Path (MSSP)}
Sometimes, routing involves multiple starting locations simultaneously. A prime example is the \textbf{Nearest Fire Station} problem: \textit{Given a city network with $K$ fire stations and multiple fire emergencies, dispatch the engine from the station with the absolute shortest travel time to the fire.}

\subsection{Super Source Simulation}
\textbf{Core Idea:} Instead of running a Single-Source Shortest Path (SSSP) algorithm separately for every fire station ($K$ times), which is heavily inefficient, we can reduce the problem mathematically by introducing a \textbf{"Super Source"} node.

\begin{enumerate}
    \item Create a virtual \textbf{Super Source} node ($S$).
    \item Add directed edges with $0$ weight from $S$ to all $K$ fire station nodes.
    \item Run standard Dijkstra's algorithm just \textbf{once} starting from $S$.
\end{enumerate}

This traversal naturally expands outward from all fire stations simultaneously. By maintaining an \texttt{origin} tracking array, the algorithm can easily trace back and identify the exact originating fire station for any intersection.

\textbf{MSSP Pseudocode:}
\begin{algorithm}[H]
\caption{MSSP - Super Source Dijkstra}
\begin{algorithmic}[1]
\Function{mssp}{$graph, sources$}
    \State $S \gets \text{new virtual node } (V)$
    \For{$src$ \textbf{in} $sources$}
        \State $graph.\text{addEdge}(S, src, 0)$
    \EndFor
    
    \State $dist \gets [\infty, \dots, \infty]$
    \State $origin \gets [-1, \dots, -1]$ \Comment{Tracks which source reached this node first}
    \State $dist[S] \gets 0$
    \State $pq \gets \text{empty priority queue (min-heap)}$
    \State $pq.\text{insert}(\{0, S\})$
    
    \While{$pq$ \textbf{is not} empty}
        \State $u, d \gets pq.\text{poll}()$
        \If{$d > dist[u]$} \textbf{continue} \EndIf
        
        \For{$edge$ \textbf{in} $graph[u]$}
            \State $v \gets edge.to$
            \State $weight \gets edge.cost$
            \If{$dist[u] + weight < dist[v]$}
                \State $dist[v] \gets dist[u] + weight$
                \If{$u == S$}
                    \State $origin[v] \gets v$ \Comment{The source node itself}
                \Else
                    \State $origin[v] \gets origin[u]$ \Comment{Inherit the origin from parent}
                \EndIf
                \State $pq.\text{insert}(\{dist[v], v\})$
            \EndIf
        \EndFor
    \EndWhile
    
    \State \Return $dist, origin$
\EndFunction
\end{algorithmic}
\end{algorithm}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{Time Complexity: $\mathbf{O((V + E) \log V)}$}. We are performing exactly one standard Dijkstra run on a graph with $V+1$ nodes and $E+K$ edges.
    \item \textbf{Space Complexity: $\mathbf{O(V + E)}$}. Required for the graph structures, priority queue, and the \texttt{dist}/\texttt{origin} tracking arrays.
\end{itemize}

\subsubsection{Step-by-Step Demonstration: MSSP}
\begin{enumerate}
    \item \textbf{Step 1 (Super Source Creation):} A massive virtual node appears, sending 0-weight edges to all designated source nodes (e.g., the Fire Stations).
    \item \textbf{Step 2 (Simultaneous Expansion):} The priority queue is initialized with the Super Source. As the algorithm runs, the visualizer shows a "wave" of shortest path expansions radiating out from \textit{all} source nodes at the exact same time.
    \item \textbf{Step 3 (Territory Claiming):} When a normal node is visited and its distance is finalized, it is color-coded to match the specific Fire Station that reached it first. Over time, this draws Voronoi-like routing zones across the graph.
\end{enumerate}
